USE mountzinai ;

CREATE TABLE admissions (
  ID VARCHAR(10) NOT NULL,
  PatientNIC VARCHAR(15) NOT NULL,
  BedNumber INT NOT NULL,
  RoomNumber INT NOT NULL,
  AdmitDate DATETIME NOT NULL,
  DischargeDate DATETIME NOT NULL,
  PRIMARY KEY (ID));
 
CREATE TABLE appointment (
  AppointmentID VARCHAR(10) NOT NULL,
  PatientNIC VARCHAR(15) NOT NULL,
  PatientName VARCHAR(100) NULL DEFAULT NULL,
  DocName VARCHAR(100) NULL DEFAULT NULL,
  DocID VARCHAR(15) NOT NULL,
  Date DATE NULL DEFAULT NULL,
  Time TIME NULL DEFAULT NULL,
  RoomNu INT NULL DEFAULT NULL,
  PRIMARY KEY (AppointmentID));
 
CREATE TABLE doctor (
  NIC VARCHAR(15) NOT NULL,
  DocID VARCHAR(15) NOT NULL,
  FullName VARCHAR(100) NOT NULL,
  NameWithInitials VARCHAR(100) NOT NULL,
  AddressL1 VARCHAR(255) NOT NULL,
  AddressL2 VARCHAR(255) NULL DEFAULT NULL,
  City VARCHAR(45) NOT NULL,
  PostalCode VARCHAR(10) NOT NULL,
  ContactNumber VARCHAR(10) NOT NULL,
  Email VARCHAR(100) NOT NULL,
  Gender VARCHAR(10) NOT NULL,
  Qualifications LONGTEXT NOT NULL,
  Specialization VARCHAR(100) NOT NULL,
  Password VARCHAR(255) NOT NULL,
  PRIMARY KEY (NIC));
 
CREATE TABLE medicalreport (
  NIC INT NOT NULL,
  ReportNumber VARCHAR(45) NOT NULL,
  TimeStamp TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  ReportContent LONGTEXT NULL DEFAULT NULL,
  Attachments LONGTEXT NULL DEFAULT NULL,
  PRIMARY KEY (NIC, ReportNumber));
 
CREATE TABLE patient (
  NIC VARCHAR(15) NOT NULL,
  PID VARCHAR(15) NOT NULL,
  RegistrationDate DATETIME NOT NULL,
  FullName VARCHAR(100) NOT NULL,
  NameWithInitials VARCHAR(100) NOT NULL,
  DOB DATETIME NOT NULL,
  AddressL1 VARCHAR(255) NOT NULL,
  AddressL2 VARCHAR(255) NULL DEFAULT NULL,
  City VARCHAR(45) NOT NULL,
  PostalCode VARCHAR(10) NOT NULL,
  ContactNumber VARCHAR(10) NOT NULL,
  Gender VARCHAR(10) NOT NULL,
  Height DECIMAL(10,0) NULL DEFAULT NULL,
  Weight DECIMAL(10,0) NULL DEFAULT NULL,
  BloodType VARCHAR(5) NULL DEFAULT NULL,
  OtherDetails LONGTEXT NULL DEFAULT NULL,
  Email VARCHAR(100) NOT NULL,
  Password VARCHAR(255) NOT NULL,
  PRIMARY KEY (NIC));
 
CREATE TABLE room (
  roomNumber INT NOT NULL,
  PRIMARY KEY (roomNumber));
 
CREATE TABLE session (
  DocID VARCHAR(15) NOT NULL,
  RoomNumber INT NOT NULL,
  Date DATE NOT NULL,
  StartTime TIME NOT NULL,
  EndTime TIME NOT NULL,
  PRIMARY KEY (DocID));
 
CREATE TABLE staff (
  NIC VARCHAR(15) NOT NULL,
  FullName VARCHAR(100) NOT NULL,
  NameWithInitials VARCHAR(100) NOT NULL,
  RegistartionDate DATETIME NOT NULL,
  AddressL1 VARCHAR(255) NOT NULL,
  AddressL2 VARCHAR(255) NULL DEFAULT NULL,
  City VARCHAR(45) NOT NULL,
  PostalCode VARCHAR(10) NOT NULL,
  ContactNumber VARCHAR(10) NOT NULL,
  Email VARCHAR(100) NOT NULL,
  Gender VARCHAR(10) NOT NULL,
  Qualifications LONGTEXT NULL DEFAULT NULL,
  Position VARCHAR(100) NOT NULL,
  Password VARCHAR(255) NOT NULL,
  PRIMARY KEY (NIC));
